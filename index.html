  | "TypeScript"
  | "JavaScript"
  | "HTML"
  | "CSS"
  | "Python"
  | "C"
  | "SQL"
  | "Bash";

interface DevProfile {
  name: string;
  role: string;
  languages: readonly Language[];
  toolchain: readonly string[];
  paradigms:
    readonly ("OOP" | "FP"
    | "imperative")[];
}

export const profile: DevProfile = {
  name: "차예준",
  role: "student-developer",
  languages: [

    "Python", "C", "SQL", "Bash",
  ] as const,
  toolchain: [
    "Node", "Git", "VS Code",
    "Chrome DevTools",
  ] as const,
  paradigms: ["OOP", "imperative"],

export const repoMeta = {
  stack:

소개

안녕하세요, 차예준입니다. 웹과 프로그래밍을 배우며 작은 결과물을 쌓아 가고 있고, 이 페이지는 그 과정을 정리해 두는 공간입니다. 아래 사진과 글을 통해 저를 조금 더 가깝게 느껴 주시면 좋겠습니다.

어떤 사람인가요

호기심이 많고, 한 번 궁금해지면 직접 해 보고 싶어 하는 편입니다. 수학·논리 퍼즐을 좋아해서 프로그래밍의 ‘문제를 나누어 푼다’는 과정에 자연스럽게 끌렸고, 지금은 웹 프론트엔드 쪽에 시간을 많이 쏟고 있습니다.

관심 있는 것들

화면이 어떤 구조로 나뉘는지(레이아웃), 사용자가 읽기 쉬운 타이포와 색(가독성), 그리고 모바일과 PC에서 모두 잘 보이게 만드는 반응형에 관심이 있습니다. 접근성과 성능은 아직 배워 가는 단계지만, 의식하고 코드에 반영하려고 합니다.

지금 하고 있는 일

HTML·CSS·JavaScript로 정적 페이지를 직접 구성해 보고, 폼 검증이나 간단한 DOM 조작 같은 바닐라 JS 패턴을 익히고 있습니다. Git으로 버전을 남기고, README에 의도와 실행 방법을 적는 연습도 병행 중입니다.

앞으로

작은 토이 프로젝트를 꾸준히 추가하고, 배운 것을 짧은 글이나 스니펫으로 정리해 이 사이트에 올릴 계획입니다. 취업·인턴·스터디 등 함께할 기회가 있다면 연락 주시면 감사하겠습니다.
